	\begin{algor}{DFS}
	def dfs(G, r):
		// Pilha contendo a raiz da árvore geradora
		seja S <- _pilha_({r})
		
		// Vetor de Retorno
		seja n <- |_v_(G)|
		seja R[n]
		
		enquanto |S| > 0:
			seja w <- S.remover()
			seja A <- _viz_(G, w)
			
			se |A| > 0:
				para cada u em A:
					se u.cor == 0:
						u.cor <- 1
						S.inserir(u)
						R[u] <- w
					senão:
						continua
			senão:
				continua

		retorna R
	\end{algor}
